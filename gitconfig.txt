alias.pushallans=!f() { 
    read -p "提交说明: " MSG; 
    
    # 遍历所有子模块，获取当前分支名并推送
    git submodule foreach "\
        CURRENT_BRANCH=\$(git rev-parse --abbrev-ref HEAD); \
        echo \"子模块: \$(pwd) 分支: \$CURRENT_BRANCH\"; \
        git add . && \
        (git commit -m \"$MSG\" || echo \"没有变更需要提交\") && \
        (git push origin \$CURRENT_BRANCH || echo \"推送失败，可能需要手动推送\")"
    
    # 主仓库，获取当前分支名并推送
    MAIN_BRANCH=$(git rev-parse --abbrev-ref HEAD)
    echo "主仓库分支: $MAIN_BRANCH"
    git add . && \
    (git commit -m "Update submodule references: $MSG" || echo "主仓库没有变更需要提交") && \
    git push origin $MAIN_BRANCH --recurse-submodules=on-demand
}; f
